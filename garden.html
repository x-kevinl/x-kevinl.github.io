# tree_ascii_simulator.py # Simulates growth/shrink of a tree image by resizing and converting to ASCII art # Requires pillow: pip install pillow import sys import time from PIL import Image # ASCII characters from dark to light ASCII_CHARS = ['@', '%', '#', '*', '+', '=', '-', ':', '.', ' '] def map_pixel_to_ascii(pixel_value): """Map a single grayscale pixel to an ASCII char.""" # pixel_value: 0 (black) to 255 (white) index = int(pixel_value / 255 * (len(ASCII_CHARS) - 1)) return ASCII_CHARS[index] def image_to_ascii(img, new_width=80): """Convert PIL Image to ASCII art string with given width, preserving aspect ratio.""" # convert to grayscale gray = img.convert('L') width, height = gray.size aspect_ratio = height / width # approximate factor to compensate for console char ratio new_height = int(new_width * aspect_ratio * 0.5) resized = gray.resize((new_width, new_height)) pixels = resized.getdata() # build string ascii_str = ''.join(map(map_pixel_to_ascii, pixels)) # split into lines ascii_lines = [ascii_str[i:i+new_width] for i in range(0, len(ascii_str), new_width)] return "\n".join(ascii_lines) def simulate_growth(path, scales, width_base=80, delay=1.0): """ Simulate tree growth by resizing according to scales list. - path: image file path - scales: list of floats, scale factors relative to base width - width_base: base width in characters - delay: seconds between frames """ try: img = Image.open(path) except Exception as e: print(f"Error loading image: {e}") sys.exit(1) for factor in scales: new_w = max(10, int(width_base * factor)) ascii_art = image_to_ascii(img, new_width=new_w) # clear screen print("\033[H\033[J", end="") print(ascii_art) time.sleep(delay) if __name__ == '__main__': import argparse parser = argparse.ArgumentParser(description='Simulate ASCII growth of a tree image') parser.add_argument('image_path', help='Path to the tree image, e.g. tree-2027899_1280.webp') parser.add_argument('--steps', type=int, default=5, help='Number of growth steps') parser.add_argument('--delay', type=float, default=1.0, help='Seconds between frames') args = parser.parse_args() # generate scale factors from small to full size steps = args.steps scales = [i/steps for i in range(1, steps+1)] simulate_growth(args.image_path, scales, width_base=80, delay=args.delay)
